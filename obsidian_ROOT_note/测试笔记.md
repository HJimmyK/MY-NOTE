#include <iostream>
#include <vector>
#include <fstream>
#include <bitset>
#include <cmath>
#include <chrono>

const int MAX_N = 1e9;			   // 10^9
const int BITSET_SIZE = MAX_N / 2; // 由于只需考虑偶数以外的数

// 线性欧拉筛法
void sieve()
{
	std::bitset<BITSET_SIZE> is_prime;
	std::vector<int> primes;

	// 0 和 1 不是素数
	is_prime[0] = false;

	// 欧拉筛法
	for (int i = 2; i <= MAX_N; ++i)
	{
		if (i < 2 || (i % 2 == 0 && i != 2))
		{
			continue; // 跳过非素数以及偶数
		}

		int idx = (i % 2 == 0) ? 0 : (i / 2);
		if (is_prime[idx])
		{
			primes.push_back(i);
			// 标记所有 i 的倍数为非素数
			for (long long j = (long long)i * i; j <= MAX_N; j += i)
			{
				if (j % 2 == 0)
					continue;
				is_prime.set(j / 2) ;
			}
		}
	}

	// 将结果写入文件
	// 第一个文件：标记素数
	std::ofstream prime_flags("prime_flags.bin", std::ios::binary);
	for (int i = 0; i < BITSET_SIZE; ++i)
	{
		prime_flags.put(is_prime[i] ? 1 : 0);
	}
	prime_flags.close();

	// 第二个文件：存储素数
	std::ofstream prime_list("prime_list.txt");
	for (const int &prime : primes)
	{
		prime_list << prime << "\n";
	}
	prime_list.close();
}

// 查找是否为素数
bool isPrime(int x)
{
	if (x < 2 || x > MAX_N)
		return false;

	// 读取标记文件
	std::ifstream prime_flags("prime_flags.bin", std::ios::binary);
	prime_flags.seekg(x / 2);
	char flag = prime_flags.get();
	prime_flags.close();

	return flag == 1;
}


int main()
{

	sieve(); // 进行筛选

	// 测试查找
	
	auto start_time = std::chrono::high_resolution_clock::now();

	int number_to_check = 209001; // 修改为需要检查的数字
	if (isPrime(number_to_check))
	{
		std::cout << number_to_check << " 是素数。" << std::endl;
	}
	else
	{
		std::cout << number_to_check << " 不是素数。" << std::endl;
	}
	// 获取结束时间
	auto end_time = std::chrono::high_resolution_clock::now();
	// 计算持续时间
	std::chrono::duration<double> duration = end_time - start_time;
	// 输出运行时间
	std::cout << "程序运行时间: " << duration.count() << "秒" << std::endl;
	return 0;
}
